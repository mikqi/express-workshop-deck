### Lets start

---

**Simple HTTP Server with Node**
```js
const http = require('http')

http.createServer((request, response) => {
  response.end('Hello NodeJS')
}).listen(1234, () => console.log('Server run on port 1234'))
```

**HTTP Server with Node**
```js
const http = require('http')
const requestHandler = (request, response) => {
  switch (request.url) {
    case '/':
      return response.end('Ini homepage')
    case '/profile':
      return response.end('Ini Halaman Profile!!')
    case '/about':
      return response.end('Ini Halaman About')
    default:
      return response.end('404 Page Not Found')
  }
}
http.createServer(requestHandler)
  .listen(1234, () => console.log('Server run on port 1234'))
```
